Simple Drag 'N Drop Prototype

Designed to use the html5 API to drag or move elements from the uh, repository of hubs/buttons to their proper locations.

function allowDrop(ev) {
    ev.preventDefault();
}

function drag(ev) {
    ev.dataTransfer.setData("text", ev.target.id);
}

function drop(ev) {
    ev.preventDefault();
    var data = ev.dataTransfer.getData("text");
    ev.target.appendChild(document.getElementById(data));
}
